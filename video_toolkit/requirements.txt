# Video Toolkit — Python Dependencies
# =====================================
# Install all dependencies:
#   pip install -r requirements.txt
#
# System dependencies (install separately):
#   - FFmpeg: https://ffmpeg.org/download.html
#   - OBS Studio: https://obsproject.com (for desktop recording)
#   - Xvfb: apt install xvfb (for headless recording)
#   - ImageMagick: apt install imagemagick (for headless capture)
#   - xdotool: apt install xdotool (for Linux window focus)

# ── Core automation ────────────────────────────────────────────────────────
pyautogui>=0.9.54       # Desktop GUI automation (mouse, keyboard)
pynput>=1.7.6           # Keyboard listener (used by calibration tool)

# ── OBS WebSocket controller ──────────────────────────────────────────────
obsws-python>=1.7.2     # OBS WebSocket 5.x client

# ── TTS narration ─────────────────────────────────────────────────────────
edge-tts>=6.1.9         # Microsoft edge-tts (free, 400+ voices)
mutagen>=1.47.0         # Audio duration detection

# ── Diagram generation ────────────────────────────────────────────────────
playwright>=1.42.0      # Headless Chromium for PNG rendering
# After installing: playwright install chromium

# ── Video assembly ────────────────────────────────────────────────────────
# FFmpeg is required (not a Python package — install separately):
#   Windows: https://ffmpeg.org/download.html
#   Linux:   apt install ffmpeg
#   Mac:     brew install ffmpeg

# ── CLI framework ─────────────────────────────────────────────────────────
click>=8.1.7            # Command-line interface

# ── Optional: Unicode clipboard paste ─────────────────────────────────────
